\section{Question 12}
\horz
Compute 
\begin{align*}
\min_{a,b,c \in\mathbb R} \int_{-1}^1|x^3-a-bx-cx^2|^2 dx
\end{align*}
and find 
\begin{align*}
\max_{g\in S} \int_{-1}^1 x^3g(x) dx,
\end{align*}
where $S= \bigg\{g\in L^2[-1,1] : \int_{-1}^{1} g(x)dx= \int_{-1}^{1} xg(x)dx = \int_{-1}^{1} x^2g(x)dx =0, \int_{-1}^{1} |g(x)|^2dx=1 \bigg\}.$
\horz
